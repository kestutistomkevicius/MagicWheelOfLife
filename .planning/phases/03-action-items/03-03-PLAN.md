```typescript
import { supabase } from '@/lib/supabase'
import type { CategoryRow } from '@/types/database'

export interface UseCategoriesResult {
  addCategory: (params: { ... }) => Promise<CategoryRow | { error: string }>
  renameCategory: (params: { ... }) => Promise<void>
  removeCategory: (params: { ... }) => Promise<void | { error: string }>
}

export function useCategories(): UseCategoriesResult {
  async function addCategory(params) {
    if (currentCount >= 12) return { error: 'Maximum 12 categories reached' }
    const res = await supabase.from('categories').insert({...}).select()
    const rows = Array.isArray(res.data) ? (res.data as CategoryRow[]) : []
    const newCategory = rows[0]
    if (!newCategory || res.error) return { error: res.error?.message ?? 'Failed to add category' }
    return newCategory
  }
  return { addCategory, renameCategory, removeCategory }
}
```
